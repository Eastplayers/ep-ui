## Install

```bash
npm install --save ep-ui
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'ep-ui'
import 'ep-ui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
